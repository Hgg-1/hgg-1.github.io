JavaScript核心语法学习

本教程涵盖了JavaScript基础语法，包括变量声明、数据类型、运算符、流程控制、循环结构和函数定义等核心概念

代码示例

// 变量声明与类型
let score = 100;
const apiKey = 'secure-key-12345';
console.log(score);
console.log(typeof score, typeof apiKey, typeof true);

let strNum = '1';
console.log(score + strNum); // 字符串拼接
console.log(score + Number(strNum)); // 数值相加

let strScore = '100';
console.log(score == strScore); // 宽松相等
console.log(score === strScore); // 严格相等
// 条件语句
let currentTime = '12:20';
if (currentTime === '12:20') {
  console.log('午餐时间到了！');
} else {
  console.log('继续学习...');
}
// 循环结构
for (let i = 1; i <= 5; i++) {
  console.log('计数: ' + i);
}

let energy = 60;
while (energy > 0) {
  console.log('剩余能量: ' + energy);
  energy -= 15;
}
// 函数定义与使用
function greet() {
  console.log('你好，欢迎学习JavaScript！');
}
greet();

function showMessage(content) {
  console.log('消息内容: ' + String(content));
}
showMessage('这是传入的参数');

function createMessage(prefix, suffix) {
  return prefix + ' 核心语法 ' + suffix;
}
let message = createMessage('JavaScript', '学习');
console.log(message);

变量与数据类型

JavaScript 使用 let 和 const 声明变量：

  • let - 声明可重新赋值的变量
  • const - 声明常量，不可重新赋值
  • 基本数据类型: number, string, boolean, null, undefined

运算符与类型转换

JavaScript 支持多种运算符：

  • 算术运算符: +, -, *, /
  • 比较运算符: ==, ===, !=, !==
  • 逻辑运算符: &&, ||, !
  • 类型转换: String(), Number(), Boolean()

流程控制

条件语句用于控制程序流程：

  • if...else - 基本条件判断
  • switch - 多分支选择
  • 三元运算符 - 简洁的条件表达式

循环结构

循环用于重复执行代码块：

  • for - 指定次数的循环
  • while - 条件为真时循环
  • do...while - 至少执行一次的循环
  • break 和 continue - 控制循环流程

函数

函数是可重用的代码块：

  • 使用 function 关键字定义
  • 可以接受参数
  • 使用 return 返回值
  • 函数不调用不会执行

作用域与闭包

理解变量作用域：

  • 全局作用域 - 整个程序可访问
  • 函数作用域 - 函数内部可访问
  • 块级作用域 - {} 内部可访问 (ES6)
  • 闭包 - 函数访问其词法作用域外变量的能力
